🎯 REDIRECTING CONSTRUCTORS
Mengimplementasikan konsep Redirecting Constructors pada kelas Circle. Konstruktor utama menerima parameter radius dengan validasi input ketat. Tiga konstruktor pengarah — fromDiameter, fromKeliling, dan fromLuas — mengalihkan instansiasi ke konstruktor utama tanpa body terpisah (menerapkan prinsip DRY).
📄 lib/circle.dart

import 'dart:math' as math;

class Circle {
  final double radius;

  Circle(this.radius) {
    if (radius <= 0) {
      throw ArgumentError('✨ Jari-jari lingkaran harus bernilai positif! ✨');
    }
  }

  Circle.fromDiameter(double d) : this(d / 2.0);

  Circle.fromKeliling(double circumference) : this(circumference / (2.0 * math.pi));

  Circle.fromLuas(double area) : this(math.sqrt(area / math.pi));

  double get diameter => radius * 2;
  double get keliling => 2.0 * math.pi * radius;
  double get luas => math.pi * radius * radius;

  @override
  String toString() {
    return 'Circle(radius: ${radius.toStringAsFixed(2)}, '
        'diameter: ${diameter.toStringAsFixed(2)}, '
        'keliling: ${keliling.toStringAsFixed(2)}, '
        'luas: ${luas.toStringAsFixed(2)})';
  }
}

import 'libcircle.dart';

void main() {
  print('━━━ REDIRECTING CONSTRUCTORS DEMO ━━━');
  print('M.ALGERIJA HERTYAZ PASLAM | 251410021 | SI2A\n');

  print('🔵 Circle dari radius (5.0)');
  final c1 = Circle(5.0);
  print('   $c1\n');

  print('🟢 Circle dari diameter (10.0) → radius = 5.0');
  final c2 = Circle.fromDiameter(10.0);
  print('   $c2');
  print('   ✅ radius sama? ${c1.radius == c2.radius}\n');

  print('🟠 Circle dari keliling (~31.416)');
  final c3 = Circle.fromKeliling(31.41592653589793);
  print('   $c3');
  print('   ✅ radius ≈ 5.0? ${(c1.radius - c3.radius).abs() < 1e-9}\n');

  print('🔴 Circle dari luas (~78.54)');
  final c4 = Circle.fromLuas(78.53981633974483);
  print('   $c4');
  print('   ✅ radius ≈ 5.0? ${(c1.radius - c4.radius).abs() < 1e-9}\n');

  print('⚠️  Validasi Error: radius negatif');
  try {
    Circle(-2.0);
  } catch (e) {
    print('   ❌ ${e.toString().split('(')[0]}');
  }

  print('\n⚠️  Validasi Error: diameter nol');
  try {
    Circle.fromDiameter(0);
  } catch (e) {
    print('   ❌ ${e.toString().split('(')[0]}');
  }

  print('\n✨ Demo selesai ✨');
}
