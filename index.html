import 'dart:math' as math;

class Circle {
  final double radius;

  // primary constructor
  Circle(this.radius) {
    if (radius <= 0) throw ArgumentError('!! RADIUS MUST BE POSITIVE !!');
  }

  // redirecting : from diameter
  Circle.fromDiameter(double d) : this(d / 2.0);

  // redirecting : from keliling (circumference)
  Circle.fromKeliling(double circ) : this(circ / (2.0 * math.pi));

  // redirecting : from luas (area)
  Circle.fromLuas(double area) : this(math.sqrt(area / math.pi));

  double get diameter => radius * 2;
  double get keliling => 2.0 * math.pi * radius;
  double get luas => math.pi * radius * radius;

  @override
  String toString() => '[CIRCLE] r=${radius.toStringAsFixed(3)} | d=${diameter.toStringAsFixed(3)} | k=${keliling.toStringAsFixed(3)} | L=${luas.toStringAsFixed(3)}';
}
